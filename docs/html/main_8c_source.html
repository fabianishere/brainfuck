37 int main()
38 {
39  /* Declaration */
40  struct BrainfuckScript *script;
41  struct BrainfuckEngineContext ctx;
42  int error;
43  FILE *file;
44 
45  /* Initialization */
46  ctx.read = &getchar;
47  ctx.write = &putchar;
48  ctx.memory = calloc(30000, sizeof(unsigned char));
49  file = fopen("../examples/hanoi.bf", "r");
50 
51  /* Parsing */
52  script = brainfuck_parser_parse_file(file, &error);
53  /*script = brainfuck_parser_parse_string("+[]", &error);*/
54 
55  if (error != BRAINFUCK_EOK) {
56  fprintf(stderr, "error: parser exited with code %i.\n", error);
57  fclose(file);
58  free(ctx.memory);
59  return 1;
60  }
61 
62  /* Passes */
63  brainfuck_pass_clear(script);
64 
65  /* Execution */
66  error = brainfuck_engine_run(script, &ctx);
67  /*brainfuck_script_dump(script, stdout);*/
68 
69  /* Cleaning */
70  fclose(file);
71  free(ctx.memory);
72  brainfuck_script_dealloc(script);
73 
74  /* Exit */
75  return error != BRAINFUCK_EOK;
76 }
